def insertion_sort_e(A):
    for j in range(1, len(A)):                     #| c1 | n
        key = A[j]                                 #| c2 | n - 1
        i = j - 1                                  #| c3 | n - 1
        while i >= 0 and A[i] > key:               #| c4 | sum_j=2 > n(t_j)

        A[i + 1] = key                             #| c7 | n - 1

def insertion_sort(A):
    for j in range(1, len(A)):
        key = A[j]
        i = j - 1
        while i >= 0 and A[i] > key:
            A[i + 1] = A[i]
            i -= 1
        A[i + 1] = key
